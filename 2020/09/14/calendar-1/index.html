打印日历的C语言实现



实现nextDate函数，通过调用nextDate函数来打印出2021年的日历

int nextDate(int month, int date, int year)
要求打印出指定日期的下一天，即给出某年某月某日，需返回下一天是多少号。
具体代码如下：

//***********************************************
//*** a program to print the calendar of 2021 ***
//***********************************************

#include <stdio.h>

//this is a function to return the next date of the given day
int nextDate(int month, int date, int year){

    //use two arrays to store the days of leap year and common year
    int leap_days[13] = {0, 31, 29, 31, 30, 31, 30, 31, 31, 30, 31, 30, 31};
    int common_days[13] = {0, 31, 28, 31, 30, 31, 30, 31, 31, 30, 31, 30, 31};

    //at first, decide wether it is a leap year
    if((year % 4 == 0 && year % 100 != 0) || (year % 400 == 0)){
        if(date + 1 > leap_days[month]){
    //if month is December, it means next day is the first day of new year
            if(month == 12){
                year++; month++; date = 1;
            }
            else{
                date++;
            }
        }
        else{
            date++;
        }
    }
    else{
        if(date + 1 > common_days[month]){
            if(month == 12){
                year++; month++; date = 1;
            }
            else{
                date++;
            }
        }
        else{
            date++;
        }
    }
    return date;
}


//this is a function to print the calendar of 2021
void printCalendar(){
    printf("\n***********************************\n");
    printf("* WELCOME TO THE CALENDAR OF 2021 *\n");
    printf("***********************************\n\n");

    //the number of each month
    int days[13] = {0, 31, 28, 31, 30, 31, 30, 31, 31, 30, 31, 30, 31};  
    //pointers of months'name
    char* months[13] = {"fault", "January", "February", "March", "April", "May", "June", "July", "August", "September", "October", "November", "December"};

    for(int i = 1; i <= 12; i++){
        printf("***** %s *****\n", months[i]);
        printf("Sun Mon Tue Wed Tus Fri Sat\n");

        static int blanks = 5;   //a variable to count the number of the blanks 
        for(int k = 0; k < blanks; k++){
            printf("    ");
        }

        for(int j = 0; j < days[i]; j++){
            blanks++; 

            if(j < 9){
                printf("%d   ", nextDate(i, j, 2021)); //print blanks to aline
            }
            else{
                printf("%d  ", nextDate(i, j, 2021));
            }
            //this block is to decide whether to reprint the week
            if(blanks == 7) {
                printf("\n");
                blanks = 0;
            }
        }
        printf("\n\n");
    }
}

int main(){
    printCalendar();
    return 0;
}
